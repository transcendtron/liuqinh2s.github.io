public class Fibonacci {
    private int n;

    //斐波那契数列的递归求法时间复杂度是：O(2^n)
    public int RecurseFibonacci(int n) {
        if (n <= 1) {
            return n;
        }
        return RecurseFibonacci(n - 1) + RecurseFibonacci(n - 2);
    }

    //斐波那契数列的迭代求法时间复杂度是：O(n)
    public int IterationFibonacci(int n) {
        if(n==0){
            return n;
        }
        int x = 1;
        int y = 1;
        for (int i = 2; i < n; i++) {
            int temp = x;
            x = y;
            y = temp + x;
        }
        return y;
    }

    //斐波那契数列的矩阵求法的时间复杂度是：O(logn)


    public static void main(String []args){
        Fibonacci fibonacci = new Fibonacci();
        int num1 = fibonacci.RecurseFibonacci(10);
        int num2 = fibonacci.IterationFibonacci(10);
        System.out.println(num2);
    }
}
